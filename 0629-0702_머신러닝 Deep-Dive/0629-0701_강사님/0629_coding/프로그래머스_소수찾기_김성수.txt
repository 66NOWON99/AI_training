def solution(numbers):
    answer = 0
    lst = []
    lst2 = []
    cnt =0 
    tmp=[]
    for i in range(1,len(numbers)+1):
        lst.append(set(permutations(numbers,i)))
    
    #print("1 = ",lst)
    for i in lst:
        lst2.append(list(i))
    lst.clear()  
    #print("2 = ",lst2)
    for i in range(len(lst2)):
        for j in range(len(lst2[i])):
            if(lst2[i][j][0] is not '0'):
                lst.append(lst2[i][j])
            
    while True:
        if(cnt == len(lst)):
            break
        tmp = list(lst[cnt])
        answer+=check(tmp)
        tmp.clear()
        cnt+=1
    return answer
