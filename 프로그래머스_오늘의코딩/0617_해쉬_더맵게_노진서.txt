import heapq

def solution(scoville, K):
    answer = 0
    
    scoville.sort()
    if scoville[0] ==0 & scoville[1]==0 :
        return -1
    
    heap=scoville
    heapq.heapify(heap)
    
    
    while heap[0] < K :
        min1 = heapq.heappop(heap)
        min2 = heapq.heappop(heap)    
        heapq.heappush(heap, min1 + min2*2)
        answer += 1
    
    if heap[0]<K & len(heap)==1 :
        return -1
    return answer
