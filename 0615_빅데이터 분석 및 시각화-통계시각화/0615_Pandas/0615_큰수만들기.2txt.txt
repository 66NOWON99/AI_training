#순서가 바뀌지 않음
#순서가 바뀜을 고려해서 만들었음

def solution(number, k):
    answer = ''
    
    numSet=set(number)
    
    numList=list(numSet)
    numList.sort(reverse=True)
    
    dict={}
    for i in numList:
        dict[i]=number.count(i)
        # print("num:",i,"val:",dict[i])
        
    count=len(number)-k
    for j in numList:
        if(count>0):
            if(dict[j]>0):
                answer=answer+j
                count-=1
                dict[j]=dict[j]-1
    print(answer)
        


    return answer
