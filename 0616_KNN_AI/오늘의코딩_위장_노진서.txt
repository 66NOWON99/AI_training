def solution(clothes):
    answer = 0
    # print(clothes)
    
    dic={}
    for i in clothes:
        if i[1] not in dic:
            dic[i[1]]=1
        else:
            dic[i[1]]=dic.get(i[1]) + 1

    
    answer=1
    for j in dic.values():
        answer=answer * (j+1)
        print ("j",j+1)
        
    answer-=1

    
    return answer
