import math
def solution(progresses, speeds):
    answer = []
    list_day = []
    cnt = 1
    ##기능을 개발하는데 얼마나 걸리는지를 계산하여 값을 리스트로 받는다
    for pr, sp in zip(progresses, speeds):
        day = (100-pr)/sp
        day = math.ceil(day)
        list_day.append(day)
        print(list_day)
    ##각 기능들이 걸리는 시간을 비교하여 결과값을 나타내준다.
    ##try, except는 마지막 인덱스 값은 인덱스에 +1을 할 수 없어 인덱스 에러가 나오므로
    ##인덱스 에러 시 결과값으로 넣어주기 위해 사용했다.
    for i, day in enumerate(list_day):
        try:
            if day >= list_day[i+1]:
                cnt += 1
                list_day[i+1] = day
                #print('if')
                #print(answer)
            else:
                answer.append(cnt)
                cnt = 1
                #print('else')
                #print(answer)
        except IndexError:
            answer.append(cnt)
            break
    return answer

progresses = [93,30,55]
speeds = [1,30,5]
solution(progresses, speeds)
